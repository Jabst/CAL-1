137  double getEdgeCost(int vertIndex1, int vertIndex2)
138  {
139  if (vertIndex1>vertIndex2)
140  {
141  //swap
142  vertIndex1 ^= vertIndex2;
143  vertIndex2 ^= vertIndex1;
144  vertIndex1 ^= vertIndex2;
145  }
146 
147  vertIndex2-=(vertIndex1+1);
148 
149  return edges[vertIndex1][vertIndex2];
150  }
